249:const useLanguage = () => {

255:const LanguageProvider = ({ children }: { children: ReactNode }) => {
256:  const [language, setLanguageState] = useState<Language>(() => {
257:    try {
258:      const saved = localStorage.getItem('lbbc_language');
259:      if (saved === 'en' || saved === 'ar') return saved;
260:    } catch (e) {
261:      console.error('Error accessing localStorage:', e);
262:    }
263:    return 'en';
264:  });
265:
266:  const setLanguage = (lang: Language) => {
267:    setLanguageState(lang);
268:    localStorage.setItem('lbbc_language', lang);
269:    document.body.dir = lang === 'ar' ? 'rtl' : 'ltr';
270:    document.body.lang = lang;
271:  };
272:
273:  useEffect(() => {
274:    document.body.dir = language === 'ar' ? 'rtl' : 'ltr';
275:    document.body.lang = language;
276:  }, [language]);
277:
278:  const t = translations[language];
279:
280:  return (
281:    <LanguageContext.Provider value={{ language, setLanguage, t }}>
282:      {children}
283:    </LanguageContext.Provider>
